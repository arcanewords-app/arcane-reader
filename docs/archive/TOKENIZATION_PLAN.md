### 2. Р Р°СЃС‡РµС‚ С‚РѕРєРµРЅРѕРІ РїРµСЂРµРґ РїРµСЂРµРІРѕРґРѕРј

```typescript
const TOKENS_PER_10K_CHARS = {
  analysis: 2000,
  translation: 10000,
  editing: 13000,
  total: 25000,
};

function estimateTokensForTranslation(
  textLength: number,
  options: { skipAnalysis?: boolean; skipEditing?: boolean }
): number {
  const charsIn10K = textLength / 10000;
  let tokens = 0;

  if (!options.skipAnalysis) tokens += TOKENS_PER_10K_CHARS.analysis * charsIn10K;
  tokens += TOKENS_PER_10K_CHARS.translation * charsIn10K;
  if (!options.skipEditing) tokens += TOKENS_PER_10K_CHARS.editing * charsIn10K;

  return Math.ceil(tokens);
}
```

## РњРёРіСЂР°С†РёСЏ СЃСѓС‰РµСЃС‚РІСѓСЋС‰РёС… РґР°РЅРЅС‹С…

### РђРіСЂРµРіР°С†РёСЏ РёСЃС‚РѕСЂРёС‡РµСЃРєРёС… РґР°РЅРЅС‹С…

Р•СЃР»Рё РЅСѓР¶РЅРѕ Р·Р°РїРѕР»РЅРёС‚СЊ `user_token_usage` РґР°РЅРЅС‹РјРё РёР· СЃСѓС‰РµСЃС‚РІСѓСЋС‰РёС… РїРµСЂРµРІРѕРґРѕРІ:

```sql
-- РњРёРіСЂР°С†РёСЏ: Р°РіСЂРµРіР°С†РёСЏ С‚РѕРєРµРЅРѕРІ РёР· chapters.translation_meta
INSERT INTO user_token_usage (user_id, date, tokens_used, tokens_by_stage)
SELECT
  p.user_id,
  DATE(c.translation_meta->>'translatedAt') as date,
  SUM((c.translation_meta->>'tokensUsed')::INTEGER) as tokens_used,
  jsonb_build_object(
    'analysis', SUM(COALESCE((c.translation_meta->'tokensByStage'->>'analysis')::INTEGER, 0)),
    'translation', SUM(COALESCE((c.translation_meta->'tokensByStage'->>'translation')::INTEGER, 0)),
    'editing', SUM(COALESCE((c.translation_meta->'tokensByStage'->>'editing')::INTEGER, 0))
  ) as tokens_by_stage
FROM chapters c
JOIN projects p ON c.project_id = p.id
WHERE c.translation_meta IS NOT NULL
  AND c.translation_meta->>'translatedAt' IS NOT NULL
GROUP BY p.user_id, DATE(c.translation_meta->>'translatedAt')
ON CONFLICT (user_id, date) DO UPDATE SET
  tokens_used = EXCLUDED.tokens_used,
  tokens_by_stage = EXCLUDED.tokens_by_stage;
```
